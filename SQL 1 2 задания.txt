1 задание
SELECT C.login,
COUNT(*) 
FROM "Couriers" AS C
INNER JOIN "Orders" AS O ON O."CourierId" = c.id
WHERE O. "inDelivery" = true
GROUP BY C.login;

2 задание

SELECT O.track,
CASE WHEN O."finished" = true THEN 2
WHEN O."cancelled" = true THEN -1
WHEN O."inDelivery" = true THEN 1
ELSE 0
END
FROM "Orders" AS O;
